# GSH v2.6.0 — Daily Delinquent Summary Email Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Send a daily HTML email to `admin@stunna.xyz` at 4:10 AM Eastern listing every delinquent subscriber with a website link and a Telegram deep link that triggers the existing inline payment button.

**Architecture:** New asyncio background task `run_daily_summary()` added to `main.py` lifespan (same pattern as the existing heartbeat). It wakes at 4:10 AM US/Eastern, queries all delinquent subscribers, builds an HTML email with one card per subscriber, and sends it via Gmail SMTP. Each card has a "View on Website" link and a "Record via Telegram" deep link (`t.me/BOT?start=pay_{acc_id}`). The Telegram webhook already handles inbound messages — we add a `/start pay_*` branch that sends the existing due-notice inline button message directly to the caller's chat.

**Tech Stack:** Python 3.11 (`zoneinfo` built-in for timezone), `smtplib` + `email.mime` (stdlib, no new deps), FastAPI asyncio lifespan, existing Gmail SMTP config in `.env`

```python
def notify_email_html(subject: str, html_body: str, to: str) -> bool:
    cfg = CONFIG["NOTIFICATIONS"]["EMAIL"]
    if not cfg.get("enabled"):
        return False
    username = cfg.get("username")
    password = cfg.get("password")
    if not username or not password:
        logger.warning("notify_email_html: email credentials not configured")
        return False
    from_email = cfg.get("from_email", "billing@guardianstreams.com")
    from_name = cfg.get("from_name", "GuardianStreams")
    smtp_server = cfg.get("smtp_server", "smtp.gmail.com")
    smtp_port = int(cfg.get("smtp_port", 587))
    try:
        msg = MIMEMultipart("alternative")
        msg["Subject"] = subject
        msg["From"] = f"{from_name} <{from_email}>"
        msg["To"] = to
        msg.attach(MIMEText(
            "Open this email in an HTML-capable client to view the GSH daily summary.",
            "plain",
        ))
        msg.attach(MIMEText(html_body, "html"))
        with smtplib.SMTP(smtp_server, smtp_port) as server:
            server.ehlo()
            server.starttls()
            server.login(username, password)
            server.sendmail(from_email, to, msg.as_string())
        logger.info("notify_email_html: sent '%s' to %s", subject, to)
        return True
    except Exception as exc:
        logger.warning("notify_email_html error: %s", exc)
        return False
```

## Task 3: `daily_summary.py` — `send_daily_summary()` + asyncio scheduler

- [ ] **Step 1: Append `send_daily_summary()` and `run_daily_summary()` to `web/backend/daily_summary.py`**

```python

def send_daily_summary() -> None:
    """Query delinquent subscribers, build HTML, and send the summary email."""
    from routers.telegram import _bot_username  # lazy import to avoid circular dep

    result = get_all_subscribers(status_filter="delinquent", page_size=1000)
    subscribers = result["subscribers"]
    # Sort most overdue first (days_until_due most negative = longest overdue)
    subscribers.sort(key=lambda s: s["days_until_due"] if s["days_until_due"] is not None else 0)

    today = datetime.now(EASTERN).strftime("%B %d, %Y")
    count = len(subscribers)
    subject = f"GSH Daily Summary — {count} Delinquent Account{'s' if count != 1 else ''} — {today}"
    html = _build_html(subscribers, today, _bot_username)

    ok = notify_email_html(subject, html, ADMIN_EMAIL)
    if ok:
        logger.info("Daily summary sent: %d delinquent account(s)", count)
    else:
        logger.warning("Daily summary: email send failed (check EMAIL config in .env)")


async def run_daily_summary() -> None:
    """Asyncio background task: fire send_daily_summary() at 4:10 AM Eastern every day."""
    logger.info("Daily summary scheduler started (target: %02d:%02d ET)", SEND_HOUR, SEND_MINUTE)
    while True:
        now = datetime.now(EASTERN)
        target = now.replace(hour=SEND_HOUR, minute=SEND_MINUTE, second=0, microsecond=0)
        if now >= target:
            target += timedelta(days=1)
        delay = (target - now).total_seconds()
        logger.info(
            "Daily summary: next send in %.1fh at %s ET",
            delay / 3600,
            target.strftime("%Y-%m-%d %H:%M"),
        )
        await asyncio.sleep(delay)

        try:
            await asyncio.to_thread(send_daily_summary)
        except asyncio.CancelledError:
            raise
        except Exception:
            logger.exception("Daily summary send failed — will retry tomorrow")

        # Sleep ~24h to land near the same time tomorrow
        # (not exactly 24h to avoid drift against the wall clock)
        await asyncio.sleep(23 * 3600 + 55 * 60)
```

```python
def _fetch_bot_username() -> str:
    """Fetch the bot's @username from Telegram getMe. Called once at module load."""
    token = CONFIG["NOTIFICATIONS"]["TELEGRAM"].get("bot_token", "")
    if not token:
        return ""
    try:
        r = requests.get(
            f"https://api.telegram.org/bot{token}/getMe",
            timeout=10,
        )
        if r.status_code == 200 and r.json().get("ok"):
            username = r.json()["result"]["username"]
            logger.info("Telegram bot username: @%s", username)
            return username
    except Exception as exc:
        logger.warning("_fetch_bot_username error: %s", exc)
    return ""
```

```python
@asynccontextmanager
async def lifespan(app_: FastAPI):
    heartbeat_task = asyncio.create_task(run_heartbeat())
    summary_task = asyncio.create_task(run_daily_summary())
    try:
        yield
    finally:
        heartbeat_task.cancel()
        summary_task.cancel()
        for t in [heartbeat_task, summary_task]:
            try:
                await t
            except asyncio.CancelledError:
                pass
```
